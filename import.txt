  <script>
    const SHEET_API_URL =
  "https://script.google.com/macros/s/AKfycbwsxB5gwLwZglPmFch4qehOOwXi7EhXK0DYJYo6xC-hE4FLXnqiKEgi9C_jysQk1ay45Q/exec";

async function fetchMenu() {
  const res = await fetch(SHEET_API_URL);
  const data = await res.json();
  const container = document.getElementById("menu-container");
  const placeholder =
    "https://placehold.jp/24/cccccc/ffffff/300x300.png?text=No+Image+Available";

  container.innerHTML = "";

  data.forEach((item) => {
    const card = document.createElement("div");
    card.className = "menu-item";

    let finalProductImg = placeholder;
    if (item.productImg && typeof item.productImg === "string") {
      finalProductImg = item.productImg.split("^")[0].trim();
    }

    card.innerHTML = `
      <div class="image-wrapper">
        <img src="${finalProductImg}" class="product-img" 
             onerror="this.onerror=null;this.src='${placeholder}';">
      </div>
      <div class="info-tray">
        <div class="brand-name">${item.brand || "Generic"}</div>
        <div class="product-name">${item.name}</div>
      </div>
    `;

    container.appendChild(card);
    card.onclick = () => showPopup(item);
  });
}

function showPopup(item) {
  const popup = document.getElementById("popup");
  const details = document.getElementById("popup-details");
  const placeholder =
    "https://placehold.jp/24/cccccc/ffffff/300x300.png?text=No+Image+Available";

  const cleanProdImg = item.productImg
    ? item.productImg.split("^")[0].trim()
    : placeholder;

  document.getElementById("popup-title").innerText = item.name;

  details.innerHTML = `
    <div style="margin-top: 20px;" class="popup-image-container">
      <img src="${cleanProdImg}" onerror="this.src='${placeholder}';">
    </div>
    <div class="popup-brand">${item.brand || ""}</div>
    <div class="popup-description">
      <strong>Description:</strong>
      <p>${item.description || "No description available."}</p>
    </div>
  `;

  popup.classList.remove("hidden");
}

document.getElementById("close-popup").onclick = () => {
  document.getElementById("popup").classList.add("hidden");
};

fetchMenu();

  </script>
